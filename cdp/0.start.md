# 启动浏览器调试
```python
        
#!/usr/bin/env python
# -*- coding:utf-8 -*-
# %%导入库
from DrissionPage import Chromium, ChromiumOptions

# 创建配置对象
co = ChromiumOptions()
co.set_argument('--remote-allow-origins=*')



# 创建浏览器对象
browser = Chromium(co)
tab = browser.latest_tab


#访问网页..
tab.get("https://newsnow.busiyi.world/")

print(tab.title)
    
# %%
```
